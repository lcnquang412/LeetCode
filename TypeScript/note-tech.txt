*** HashMap ***
const map = new Map<string, number>();
map.set("apple", 3);
map.set("banana", 5);

// 🔁 Loop through key-value pairs
for (const [key, value] of map) {
  console.log(`${key} → ${value}`);
}

// 🔁 Loop keys only
for (const key of map.keys()) {
  console.log("key:", key);
}

// 🔁 Loop values only
for (const value of map.values()) {
  console.log("value:", value);
}

*** Loop ***
- loop..of => value
- loop..in => key, index
